```python
"""用户个性化设置路由"""
from __future__ import annotations
import json
import logging
from typing import Optional
from fastapi import APIRouter, Depends
from pydantic import BaseModel
from sqlalchemy import Column, Integer, String, DateTime, Text
from sqlalchemy.orm import Session
from datetime import datetime
from models.database import Base, get_db
from models.user import User
from utils.auth import get_current_user
from utils.response import success_response, error_response

logger = logging.getLogger(__name__)
router = APIRouter()

class UserSettings(Base):
    __tablename__ = "user_settings"
    id = Column(Integer, primary_key=True, index=True)
    user_id = Column(Integer, unique=True, nullable=False, index=True)
    ai_name = Column(String(50), default="小语")
    response_style = Column(String(20), default="gentle")
    notifications_enabled = Column(Integer, default=1)
    theme = Column(String(20), default="dark")
    language = Column(String(10), default="zh")
    crisis_alert_enabled = Column(Integer, default=1)
    daily_reminder_enabled = Column(Integer, default=0)
    reminder_time = Column(String(10), default="20:00")
    updated_at = Column(DateTime, default=datetime.utcnow)

class SettingsUpdateRequest(BaseModel):
    ai_name: Optional[str] = None
    response_style: Optional[str] = None
    notifications_enabled: Optional[bool] = None
    theme: Optional[str] = None
    crisis_alert_enabled: Optional[bool] = None
    daily_reminder_enabled: Optional[bool] = None
    reminder_time: Optional[str] = None

def _settings_to_dict(s: UserSettings) -> dict:
    return {
        "ai_name": s.ai_name or "小语",
        "response_style": s.response_style or "gentle",
        "notifications_enabled": bool(s.notifications_enabled),
        "theme": s.theme or "dark",
        "language": s.language or "zh",
        "crisis_alert_enabled": bool(s.crisis_alert_enabled),
        "daily_reminder_enabled": bool(s.daily_reminder_enabled),
        "reminder_time": s.reminder_time or "20:00",
    }

@router.get("/settings", summary="获取用户设置")
async def get_settings(current_user: User = Depends(get_current_user), db: Session = Depends(get_db)):
    s = db.query(UserSettings).filter(UserSettings.user_id == current_user.id).first()
    if not s:
        s = UserSettings(user_id=current_user.id)
        db.add(s); db.commit(); db.refresh(s)
    return success_response(data=_settings_to_dict(s))

@router.put("/settings", summary="更新用户设置")
async def update_settings(req: SettingsUpdateRequest, current_user: User = Depends(get_current_user), db: Session = Depends(get_db)):
    s = db.query(UserSettings).filter(UserSettings.user_id == current_user.id).first()
    if not s:
        s = UserSettings(user_id=current_user.id)
        db.add(s)
    if req.ai_name is not None: s.ai_name = req.ai_name
    if req.response_style is not None: s.response_style = req.response_style
    if req.notifications_enabled is not None: s.notifications_enabled = int(req.notifications_enabled)
    if req.theme is not None: s.theme = req.theme
    if req.crisis_alert_enabled is not None: s.crisis_alert_enabled = int(req.crisis_alert_enabled)
    if req.daily_reminder_enabled is not None: s.daily_reminder_enabled = int(req.daily_reminder_enabled)
    if req.reminder_time is not None: s.reminder_time = req.reminder_time
    s.updated_at = datetime.utcnow()
    db.commit(); db.refresh(s)
    return success_response(data=_settings_to_dict(s))

@router.post("/settings/reset", summary="重置设置")
async def reset_settings(current_user: User = Depends(get_current_user), db: Session = Depends(get_db)):
    s = db.query(UserSettings).filter(UserSettings.user_id == current_user.id).first()
    if s:
        db.delete(s); db.commit()
    return success_response(data={"reset": True})
```

```python
"""心理知识库路由 — 直接从 emotion_knowledge.py 读取"""
from __future__ import annotations
import logging
from typing import Optional
from fastapi import APIRouter, Query
from utils.response import success_response

logger = logging.getLogger(__name__)
router = APIRouter()

@router.get("/knowledge/articles", summary="获取知识文章列表")
async def get_articles(
    category: Optional[str] = None,
    page: int = Query(default=1, ge=1),
    page_size: int = Query(default=20, ge=1, le=50),
):
    from knowledge.emotion_knowledge import KNOWLEDGE_BASE, _normalize_entry
    entries = [_normalize_entry(e) for e in KNOWLEDGE_BASE]
    if category:
        entries = [e for e in entries if e.get("category") == category]
    total = len(entries)
    start = (page - 1) * page_size
    items = entries[start: start + page_size]
    articles = [
        {
            "id": i + start + 1,
            "title": e.get("topic", ""),
            "content": (e.get("content", "") + "\n\n" + e.get("advice", "")).strip(),
            "category": e.get("category", ""),
            "read_time": max(1, len(e.get("content", "")) // 200),
            "tags": ",".join(e.get("keywords", [])[:5]),
            "view_count": 0,
            "created_at": "2025-01-01T00:00:00",
        }
        for i, e in enumerate(items)
    ]
    return success_response(data={"articles": articles, "total": total, "page": page, "page_size": page_size})

@router.get("/knowledge/articles/{article_id}", summary="获取单篇文章")
async def get_article(article_id: int):
    from knowledge.emotion_knowledge import KNOWLEDGE_BASE, _normalize_entry
    entries = [_normalize_entry(e) for e in KNOWLEDGE_BASE]
    idx = article_id - 1
    if idx < 0 or idx >= len(entries):
        from utils.response import error_response
        return error_response(code=404, msg="文章不存在")
    e = entries[idx]
    return success_response(data={
        "id": article_id,
        "title": e.get("topic", ""),
        "content": (e.get("content", "") + "\n\n" + e.get("advice", "") + "\n\n" + e.get("dialogue_example", "")).strip(),
        "category": e.get("category", ""),
        "read_time": max(1, len(e.get("content", "")) // 200),
        "tags": ",".join(e.get("keywords", [])[:5]),
        "view_count": 0,
        "created_at": "2025-01-01T00:00:00",
    })

@router.get("/knowledge/categories", summary="获取知识分类")
async def get_categories():
    from knowledge.emotion_knowledge import KNOWLEDGE_BASE
    from collections import Counter
    counts = Counter(e.get("category", "") for e in KNOWLEDGE_BASE)
    categories = [{"name": k, "count": v} for k, v in counts.items() if k]
    return success_response(data={"categories": categories})
```

```python
class MemoryCapsule(Base):

    id = Column(Integer, primary_key=True, index=True)

    title = Column(String(200), nullable=False)
    content = Column(Text, nullable=False)

    tags = Column(String(200), default="")

    created_at = Column(DateTime, default=datetime.utcnow)

class CapsuleCreateRequest(BaseModel):
    title: str = Field(..., max_length=200)

    tags: Optional[str] = ""

class CapsuleUpdateRequest(BaseModel):
    title: Optional[str] = None

    mood: Optional[str] = None
    tags: Optional[str] = None
    is_public: Optional[bool] = None

def _cap_to_dict(c: MemoryCapsule) -> dict:
    return {
        "id": c.id, "title": c.title, "content": c.content,
        "mood": c.mood, "tags": c.tags, "is_public": c.is_public,

@router.put("/memory-capsules/{capsule_id}", summary="更新记忆胶囊")
```

```python
"""工具使用记录路由"""
from __future__ import annotations
import logging
from datetime import datetime
from typing import Optional
from fastapi import APIRouter, Depends, Query
from pydantic import BaseModel
from sqlalchemy import Column, Integer, String, Float, DateTime, Text, func

class ToolUsageRecord(Base):
    __tablename__ = "tool_usage_records"
    id = Column(Integer, primary_key=True, index=True)
    user_id = Column(Integer, nullable=True, index=True)
    tool_name = Column(String(50), nullable=False)
    duration_seconds = Column(Float, default=0)
    extra_data = Column(Text, default="")
    created_at = Column(DateTime, default=datetime.utcnow, index=True)

class ToolUsageCreateRequest(BaseModel):
    tool_name: str
    duration_seconds: Optional[float] = 0
    metadata: Optional[str] = ""

@router.post("/tool-usage", summary="记录工具使用")
async def record_usage(req: ToolUsageCreateRequest, db: Session = Depends(get_db), current_user: Optional[User] = Depends(lambda: None)):
    r = ToolUsageRecord(tool_name=req.tool_name, duration_seconds=req.duration_seconds or 0, extra_data=req.metadata or "")
    db.add(r); db.commit(); db.refresh(r)
    return success_response(data={"id": r.id})

@router.get("/tool-usage/stats", summary="工具使用统计")
async def get_stats(days: int = 30, current_user: User = Depends(get_current_user), db: Session = Depends(get_db)):
    records = db.query(ToolUsageRecord).filter(ToolUsageRecord.user_id == current_user.id).all()
    by_tool: dict = {}
    for r in records:
        if r.tool_name not in by_tool:
            by_tool[r.tool_name] = {"tool_name": r.tool_name, "usage_count": 0, "total_duration_seconds": 0}
        by_tool[r.tool_name]["usage_count"] += 1
        by_tool[r.tool_name]["total_duration_seconds"] += r.duration_seconds or 0
    return success_response(data={"total_usage": len(records), "total_duration_seconds": sum(r.duration_seconds or 0 for r in records), "by_tool": list(by_tool.values())})
```
